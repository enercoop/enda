108        :param years_list: list of years for which holidays must be returned
109        :param orientation: 'rows' (default) or 'columns'. The way to orient the dataframe

136                .reset_index(drop=False)
137                .filter(["date", "public_holiday_name"])

302    @staticmethod
303    def get_public_holidays(country: str = 'FR', years_list: list[int] = None,
304                            handling_missing_year: str = 'warning') -> pd.DataFrame:
305        """
306        Return a dataframe (at max from 2000-01-01 to 2050-12-31) indicating for each day
307        whether it is a public holiday (denoted by a 1) or not (denoted by a 0)
308        :param country: country of interest
309        :param years_list: list of years for which holidays must be returned
310        :param handling_missing_year: either 'warning' (default)  or 'error' or 'silent'
311        :return: a dataFrame with a daily DatetimeIndex and a float 'public holiday' column containing
312                 1 if the day is a public holiday and 0 otherwise
313        """
314
315        if years_list is None:
316            years_list = range(2000, 2051)
317
318        # get public holidays
319        public_holidays = Holidays.get_public_holidays(country=country,
320                                                       years_list=years_list,
321                                                       handling_missing_year=handling_missing_year)
322
323        # work it out
324        public_holidays = public_holidays.set_index("date")
325        public_holidays.index = pd.to_datetime(public_holidays.index)
326        public_holidays = public_holidays[
327            ~public_holidays.index.duplicated(keep="first")
328        ]
329        public_holidays["public_holiday"] = 1
330
331        # reindex and fillna
332        start_date = datetime.date(min(years_list), 1, 1)
333        incl_end_date = datetime.date(max(years_list), 12, 31)
334        public_holidays = (
335            public_holidays
336            .reindex(pd.date_range(start_date, incl_end_date))
337            .fillna(0)
338        )
339
340        return public_holidays[["public_holiday"]]

342    @staticmethod
343    def get_extra_long_weekend(country: str = 'FR', years_list: list[int] = None,
344                               handling_missing_year: str = "warning") -> pd.DataFrame:
345        """
346        Return a dataframe (at max from 2000-01-01 to 2050-12-31) indicating for each day
347        - if the previous (resp. the next day) is a public holiday
348        AND
349        - if the current day is a friday (resp. a monday)
350        If both conditions are fulfilled then the day is denoted by a 1 (0 otherwise)
351        :param country: country of interest
352        :param years_list: the list of target years
353        :param handling_missing_year: either 'warning' (default)  or 'error' or 'silent'
354        :return: a DataFrame with a daily DatetimeIndex and an int 'extra_long_weekend' column containing 1 if
355                 the day meets the criteria described above and 0 otherwise
356        """
357
358        public_holidays = Calendar.get_public_holidays(country, years_list=years_list,
359                                                       handling_missing_year=handling_missing_year)
360        public_holidays = DatetimeFeature.split_datetime(
361            public_holidays, split_list=["dayofweek"], index=True
362        )
363
364        public_holidays["is_yesterday_day_off"] = public_holidays[
365            "public_holiday"
366        ].shift()
367        public_holidays["is_tomorrow_day_off"] = public_holidays[
368            "public_holiday"
369        ].shift(-1)
370        public_holidays["extra_long_weekend"] = 0
371
372        mondays = public_holidays[public_holidays["dayofweek"] == 0]
373        mondays_off = mondays[mondays["is_tomorrow_day_off"] == 1].index
374
375        fridays = public_holidays[public_holidays["dayofweek"] == 4]
376        fridays_off = fridays[fridays["is_yesterday_day_off"] == 1].index
377
378        extra_long_weekend_index = mondays_off.append(fridays_off)
379        extra_long_weekend_index = sorted(extra_long_weekend_index)
380
381        public_holidays.loc[extra_long_weekend_index, "extra_long_weekend"] = 1
382
383        return public_holidays[["extra_long_weekend"]]

457    @staticmethod
458    def feature_special_days(years_list: list[int] = None,
459                             freq: Union[str, pd.Timedelta] = "30min",
460                             index_name: str = 'time',
461                             handling_missing_year: str = "warning"
462                             ) -> pd.DataFrame:
463        """
464        Return a DataFrame containing all special french days: public and school holidays, lockdowns, and extra long
465            weekends
466        :param years_list: the list of years
467        :param freq: A string indicating the frequency at which to interpolate the DataFrame
468        :param index_name:A string indicating the name of the resulting index
469        :param handling_missing_year: either 'warning' (default)  or 'error' or 'silent'
470        :return:
471            A DataFrame with a DatetimeIndex at the specified frequency, and 4 float columns :
472            - The 'lockdown' column contains 1 if the timestamp is within a lockdown period and 0 otherwise
473            - The 'public_holiday' column contains 1 if the day is a public holiday and 0 otherwise
474            - The 'nb_school_areas_off' indicates the number of French zones (3 zones : A, B and C) that are in
475            school holidays
476            - The 'extra_long_weekend' contains 1 if the day is part of a long weekend (a Monday with a public holiday
477            on Tuesday, or a Friday with a public holiday in Thursday) and 0 otherwise
478        """
479
480        # get all constitutive elements of special days
481        lockdown_df = FrenchCalendar.get_lockdown(years_list=years_list)
482        public_holidays_df = FrenchCalendar.get_public_holidays(years_list=years_list,
483                                                                handling_missing_year=handling_missing_year)
484        school_areas_off_df = FrenchCalendar.get_number_school_areas_off(years_list=years_list,
485                                                                         handling_missing_year=handling_missing_year)
486        extra_long_weekend_df = FrenchCalendar.get_extra_long_weekend(years_list=years_list,
487                                                                      handling_missing_year=handling_missing_year)
488
489        # resample everything to the desired freq
490        lockdown_df = Resample.upsample_and_interpolate(
491            lockdown_df, freq=freq, method="ffill", forward_fill=True, index_name=index_name, tz_info=TZ_PARIS
492        )
493        public_holidays_df = Resample.upsample_and_interpolate(
494            public_holidays_df, freq=freq, method="ffill", forward_fill=True, index_name=index_name, tz_info=TZ_PARIS
495        )
496        school_areas_off_df = Resample.upsample_and_interpolate(
497            school_areas_off_df, freq=freq, method="ffill", forward_fill=True, index_name=index_name, tz_info=TZ_PARIS
498        )
499        extra_long_weekend_df = Resample.upsample_and_interpolate(
500            extra_long_weekend_df, freq=freq, method="ffill", forward_fill=True, index_name=index_name, tz_info=TZ_PARIS
501        )
502
503        result = pd.concat(
504            [
505                lockdown_df,
506                public_holidays_df,
507                school_areas_off_df,
508                extra_long_weekend_df,
509            ],
510            axis=1,
511            join="outer",
512        )
513
514        return result

Returns

- The 'public_holiday' column contains 1 if the day is a public holiday and 0 otherwise
- The 'nb_school_areas_off' indicates the number of French zones (3 zones : A, B and C) that are in
