 14class EndaEstimator(metaclass=abc.ABCMeta):
 15    """
 16    This interface represents a simple machine learning estimator with some universal functions.
 17    We require these functions :
 18        train : train the estimator
 19        predict : predict using the estimator
 20        get_model_params : return the hyperparameters of the model
 21
 22    To save and load instances of a class, use tools like pickle or joblib
 23    (see information for instance here: https://scikit-learn.org/stable/modules/model_persistence.html).
 24
 25    This interface is useful to create more advanced estimators based on these building blocks :
 26        EndaNormalizedEstimator : uses one of the inputs as a 'normalization variable' instead of a 'training feature'.
 27        EndaStackingEstimator : combines several estimators to create a more robust estimator
 28                                (cross-algorithm 'ensemble' method).
 29        EndaEstimatorWithFallback : an estimator that can cope with missing input in a robust way.
 30
 31    See tutorials about Python interfaces for instance here https://realpython.com/python-interface/ .
 32    """
 33    def __init__(self):
 34        # if set, _training_df stores the training dataframe (features + target)
 35        # if set, _target_name stores the target name
 36        self._training_df = None
 37        self._target = None
 38
 39    @abc.abstractmethod
 40    def train(self, df: pd.DataFrame, target_col: str):
 41        """Trains the estimator using the given data."""
 42        raise NotImplementedError
 43
 44    def fit(self, X: pd.DataFrame, y:pd.Series):
 45        """Trains the estimator using the given data."""
 46        return self.train(df=pd.concat([X, y], axis=1), target_col=y.name)
 47
 48    @abc.abstractmethod
 49    def predict(self, df: pd.DataFrame, target_col: str) -> pd.DataFrame:
 50        """Predicts and returns a dataframe with just 1 column: target_col_name"""
 51        raise NotImplementedError
 52
 53    def get_model_name(self) -> str:
 54        """Return the estimator name"""
 55        return self.__class__.__name__
 56
 57    @abc.abstractmethod
 58    def get_model_params(self) -> dict:
 59        """Return a dict with the model name and hyperparameters"""
 60        raise NotImplementedError
 61
 62    def get_loss_training(self,
 63                          scores: Optional[Union[list[str], dict[str, str]]] = None,
 64                          process_forecast_specs: Optional[Tuple[Callable, dict]] = None) -> pd.Series:
 65        """
 66         Compute the training loss, i.e. the error of the trained model on the training dataset.
 67         If not overridden (eg. in H2OEstimator), this function computes the loss on the training
 68         dataset, using scikit-learn built-in methods.
 69        :param scores: the statistics to consider. It can be a list of enda-known functions, a home-made function
 70            (if given as a dict), or RMSE if nothing is provided.
 71        :param process_forecast_specs: Optional. If given, it defines a function to apply to the forecast before
 72            calculating the loss.
 73        :return: a series that contains for each statistics the score of the model on the training set
 74        """
 75
 76        # if _training_df or _target is None, that means the model has not been trained
 77        if self._training_df is None or self._target is None:
 78            raise ValueError("The model must be trained before calling this method.")
 79
 80        # compute the prediction over the training dataset
 81        predict_on_train_set_df = self.predict(df=self._training_df.drop(columns=self._target), target_col=self._target)
 82
 83        if process_forecast_specs is not None:
 84            process_forecast_function, process_forecast_kwargs = process_forecast_specs
 85            predict_on_train_set_df = process_forecast_function(predict_on_train_set_df, **process_forecast_kwargs)
 86
 87        score_series = Scoring.compute_loss(predicted_df=predict_on_train_set_df,
 88                                            actual_df=self._training_df[self._target],
 89                                            scores=scores)
 90
 91        return score_series
 92
 93    def get_feature_importance(self) -> pd.Series:
 94        """
 95        Return the feature's importance once a model has been trained.
 96        Such a feature is usually not implemented, except for some algorithm in
 97        particular (let's say, sklearn and H2O, and not all of them)
 98        :return: a series that contain the percentage of importance for each variable
 99        """
100        raise NotImplementedError()

103class EndaNormalizedEstimator(EndaEstimator):
104    """
105    An estimator that uses one of the inputs as a linear 'normalization variable' instead of a 'training feature'.
106    The prediction is:
107    predict(X) = X[normalization_col] * predict(X without normalization_col)
108
109    predict(X without normalization_col) is made by another underlying estimator: inner_estimator.
110    """
111
112    def __init__(
113            self,
114            inner_estimator: EndaEstimator,
115            target_col: str,
116            normalization_col: str,
117            columns_to_normalize: Optional[Iterable[str]] = None,
118    ):
119        """
120        Initialize the normalized estimator
121        :param inner_estimator: the estimator that will perform predict(X without normalization_col)
122        :param normalization_col: name of the column used as a multiplier and not as a feature.
123        :param columns_to_normalize: (optional) columns in 'X without normalization_col' that must be divided
124        by normalization_col for the underlying estimator to train/predict correctly.
125        """
126
127        if not issubclass(type(inner_estimator), EndaEstimator):
128            raise TypeError(
129                f"inner_estimator's type is '{type(inner_estimator)}' : it must implement EndaEstimator."
130            )
131
132        if columns_to_normalize and normalization_col in columns_to_normalize:
133            raise ValueError(
134                f"normalisation_col '{normalization_col}'should not be in columns_to_normalize {columns_to_normalize}"
135            )
136
137        super().__init__()
138        self.inner_estimator = inner_estimator
139        self.target_col = target_col
140        self.normalisation_col = normalization_col
141        self.columns_to_normalize = columns_to_normalize
142
143    def get_model_params(self) -> dict:
144        """Return a dict with the model name and hyperparameters"""
145        return {self.get_model_name(): self.inner_estimator.get_model_params()}
146
147    def check_normalization_col(self, df: pd.DataFrame):
148        """
149        Checks that the values of the normalization col in the input DataFrame are strictly positive, otherwise
150        raise an error
151        :param df: The DataFrame to check
152        """
153
154        if self.normalisation_col not in df:
155            raise ValueError(
156                f"Normalisation_col '{self.normalisation_col}' not present in the Dataframe to normalize"
157            )
158        zeros_df = df[df[self.normalisation_col] <= 0]
159        if not zeros_df.empty:
160            raise ValueError(
161                f"Normalisation_col '{self.normalisation_col}' : zeros found\n{zeros_df}"
162            )
163
164    def normalize(self, df: pd.DataFrame) -> pd.DataFrame:
165        """
166        Checks that the normalization column has only strictly positive values, then normalizes the columns of the
167        DataFrame that are in the columns_to_normalize attribute by dividing them by the normalization column. Also
168        normalizes the target_col if it is in the DataFrame columns. Then, drops normalization_col
169        :param df: The DataFrame to normalize
170        :return: The DataFrame with columns part of columns_to_normalize or the target_col column normalized, and
171            without the normalization_col
172        """
173        self.check_normalization_col(df)
174        df_norm = df.copy(deep=True)
175
176        if self.columns_to_normalize:
177            for col in df.columns:
178                if col in self.columns_to_normalize:
179                    df_norm[col] = df_norm[col] / df[self.normalisation_col]
180
181        # always normalize the target if it is in the df (present in train mode, not in predict mode)
182        if self.target_col in df.columns:
183            df_norm[self.target_col] = (
184                    df_norm[self.target_col] / df[self.normalisation_col]
185            )
186
187        df_norm.drop(columns=self.normalisation_col, inplace=True)
188        return df_norm
189
190    def train(
191            self,
192            df: pd.DataFrame,
193            target_col: str = None,
194            drop_where_normalization_under_zero: bool = False,
195    ):
196        """
197        Normalizes the DataFrame and trains the inner estimator
198        :param df: The training DataFrame
199        :param target_col: The variable to predict
200        :param drop_where_normalization_under_zero: Whether to drop rows where normalization_col is not strictly
201            positive. Note that the normalize function will raise an error if there is still negative values in
202            normalization_col
203        """
204        if target_col and self.target_col != target_col:
205            raise ValueError(
206                f"target should be None or {self.target_col}, but given: {target_col}"
207            )
208
209        if drop_where_normalization_under_zero:
210            df = df.loc[df[self.normalisation_col] > 0, :]
211        df_norm = self.normalize(df)
212        self.inner_estimator.train(df_norm, self.target_col)
213
214        # store the training
215        self._training_df = df
216        self._target = target_col
217
218    def predict(self, df: pd.DataFrame, target_col: str = None) -> pd.DataFrame:
219        """
220        Normalizes the input DataFrame, then uses the inner estimator to make a prediction on target_col. Then,
221        multiplies the result by the normalization_col to get the final prediction
222        :param df: The forecast input DataFrame
223        :param target_col: The variable to predict
224        :return: The final prediction of the estimator
225        """
226        if target_col and self.target_col != target_col:
227            raise ValueError(
228                f"target should be None or '{self.target_col}', but given: '{target_col}'."
229            )
230
231        df_norm = self.normalize(
232            df
233        )  # error out if any value of normalization_col is <= 0
234        predict_norm = self.inner_estimator.predict(df_norm, self.target_col)
235
236        if (predict_norm.index != df.index).any():
237            raise ValueError(
238                "prediction must have the same index as given df. "
239                "Check that for self.inner_estimator, the method 'predict' conserves index."
240            )
241
242        predict = predict_norm.multiply(df[self.normalisation_col], axis="index")
243        return predict

246class EndaStackingEstimator(EndaEstimator):
247    """
248    This class serves the same purpose as the Scikit-Learn "Stacking Regressor". However since we work on
249    time-series, we need fine control on which data is passed to train the base_estimators before training
250    the final_estimator (to keep it all chronologically consistent).
251
252    Training is made this way :
253        temporarily train base_estimators on train_set[:x]
254        use base_estimators to predict on train_set[x:] -> base_predictions
255        train final_estimator on the base_predictions
256        re-train base_estimators on the full train-set
257    We have to train base_estimators twice, because if we trained them on the full set and then predict on
258    train_set[x:] for training final_estimator, we would be making a prediction on part of the data that was used to
259    train them, which would be a huge risk of over-fitting
260    """
261
262    def __init__(
263            self,
264            base_estimators: typing.Mapping[str, EndaEstimator],
265            final_estimator: EndaEstimator,
266            base_stack_split_pct: float = 0.20,
267    ):
268        """
269        Initialize the stacking estimator
270        :param base_estimators: a dict of {estimator_id -> estimator}, each estimator (there must be at least 2) must
271            be an EndaEstimator.
272        :param final_estimator: the estimator used for stacking, must also implement be an EndaEstimator.
273        :param base_stack_split_pct: the % of data used to train the base_estimators before training final_estimator
274            (this can be overwritten in the train function).
275        """
276
277        if len(base_estimators) <= 1:
278            raise ValueError(
279                f"At least 2 base_estimators are required, but given: {len(base_estimators)}"
280            )
281
282        # We store estimators in an ordered dict to make sure we always iterate over them in the same order
283        self.base_estimators = OrderedDict()
284        for estimator_id in sorted(base_estimators.keys()):
285            self.base_estimators[estimator_id] = base_estimators[estimator_id]
286
287        super().__init__()
288        self.final_estimator = final_estimator
289        self.base_stack_split_pct = base_stack_split_pct
290
291    def train(
292            self,
293            df: pd.DataFrame,
294            target_col: str,
295            base_stack_split_pct: [float, None] = None,
296    ):
297        """
298        Train base and final estimators.
299        :param df: The training dataset
300        :param target_col: The column to predict
301        :param base_stack_split_pct: If specified, will overwrite the EndaStackingEstimator attribute.
302        """
303        split_pct = (
304            base_stack_split_pct if base_stack_split_pct else self.base_stack_split_pct
305        )
306
307        # training final_estimator will temporarily train single estimators on part of the data,
308        # so it must be done before training the actual single estimators
309        self.train_final_estimator(df, target_col, split_pct)
310
311        # re-train base estimators with the full dataset
312        self.train_base_estimators(df, target_col)
313
314        # store the training
315        self._training_df = df
316        self._target = target_col
317
318    def train_final_estimator(
319            self, df: pd.DataFrame, target_col: str, split_pct: float
320    ):
321        """
322        Trains the final estimator used for stacking.
323        (Temporarily) train the single estimators with a subset of the data,
324        then apply them on the rest of the data. Use this to train the stacking estimator.
325        :param df: The DataFrame with training data
326        :param target_col: The column to predict
327        :param split_pct: The percentage of training data to use to train the base estimators. The rest of the data
328            will be used to train the final estimator based on the predictions of the base estimators
329        """
330
331        # split the training frame : ,
332        split_int = int(df.shape[0] * (1 - split_pct))
333        split_idx = df.index[split_int]
334
335        df_base_estimators = df[
336            df.index < split_idx
337            ]  # one part to train the base estimators
338        df_stacking = df[
339            df.index >= split_idx
340            ]  # the other to train the stacking estimator
341
342        if df_base_estimators.shape[0] == 0 or df_stacking.shape[0] == 0:
343            raise ValueError(
344                "The split gave an empty train set for the base estimators or the final estimator. "
345                f"Change parameter 'split_pct' (given {split_pct}) or provide a larger training set."
346            )
347
348        self.train_base_estimators(df_base_estimators, target_col)
349
350        # make predictions with these temporary base_estimators, without the target column
351        base_predictions = self.predict_base_estimators(
352            df_stacking.drop(columns=[target_col]), target_col
353        )
354
355        # add the target back, to train the final estimator
356        base_predictions[target_col] = df_stacking[target_col]
357
358        self.final_estimator.train(base_predictions, target_col)
359
360    def predict_base_estimators(
361            self, df: pd.DataFrame, target_col: str
362    ) -> pd.DataFrame:
363        """
364        Make a prediction using base estimators
365        :param df: The input DataFrame for prediction
366        :param target_col: The name of the column to predict
367        :return: a Dataframe with the prediction of each base estimator in each column (on several rows).
368            Each column's name is the estimator_id given on initialisation.
369        """
370
371        estimator_dfs = []
372        for estimator_id, estimator in self.base_estimators.items():
373            estimator_predict = estimator.predict(df, target_col)
374
375            if (estimator_predict.index != df.index).any():
376                raise ValueError(
377                    "prediction must have the same index as given df. "
378                    f"Check that for estimator with id '{estimator_id}', the method 'predict' conserves index."
379                )
380
381            estimator_predict.rename(columns={target_col: estimator_id}, inplace=True)
382            estimator_dfs.append(estimator_predict)
383
384        predict_df = pd.concat(estimator_dfs, axis=1, join="outer")
385
386        if predict_df.shape[0] != df.shape[0]:
387            raise ValueError(
388                f"Given {df.shape[0]} values to predict, but predicted {predict_df.shape[0]}"
389            )
390
391        return predict_df
392
393    def train_base_estimators(self, df: pd.DataFrame, target_col: str):
394        """
395        Train the base estimators
396        :param df: The training data
397        :param target_col: The column to predict
398        """
399        for _, estimator in self.base_estimators.items():
400            estimator.train(df, target_col)
401
402    def predict(self, df: pd.DataFrame, target_col: str) -> pd.DataFrame:
403        """
404        Make predictions with base estimators, then uses them to make a prediction with the final estimator
405        :param df: The input data for prediction
406        :param target_col: The column to predict
407        :return: A DataFrame with the predicted column
408        """
409        base_predictions = self.predict_base_estimators(df, target_col)
410        prediction = self.final_estimator.predict(base_predictions, target_col)
411
412        if (prediction.index != df.index).any():
413            raise ValueError(
414                "prediction must have the same index as given df. "
415                "Check that self.final_estimator.predict conserves index."
416            )
417
418        return prediction
419
420    def get_model_params(self) -> dict:
421        """
422        Get model parameters of estimators (each of base_estimator) and final_estimator
423        The principle is to return the model params as  a dict with two entries, base_estimators, and final_estimator
424        Each of these entries contains a dictionary as well, with the model name as a key, and model parameters
425        as values.
426        :return: A dictionary with one entry per model and associated parameters
427        """
428
429        model_params_dict = {"base_estimators": collections.defaultdict(dict)}
430
431        # define base estimator
432        for _, estimator in self.base_estimators.items():
433            model_params = estimator.get_model_params()
434            for estimator_name, estimator_params in model_params.items():
435
436                # we need to modify the estimator name in the dict keys, if the same estimator
437                # is defined several times. For instance, if several LinearRegression are defined,
438                # we call them LinearRegression, LinearRegression_1, LinearRegression_2...
439                original_estimator_name = estimator_name
440                count = 1
441                while estimator_name in model_params_dict["base_estimators"]:
442                    estimator_name = f"{original_estimator_name}_{count}"
443                    count += 1
444                model_params_dict["base_estimators"][estimator_name] = estimator_params
445        model_params_dict["base_estimators"] = dict(model_params_dict["base_estimators"])
446        model_params_dict["final_estimator"] = self.final_estimator.get_model_params()
447
448        return {self.get_model_name(): model_params_dict}
449
450    def get_all_model_names(self) -> dict[str, list[str]]:
451        """
452        Get name of all base models, plus name of final model
453        in a dictionary with two entries, "base_estimator" and "final_estimator"
454        """
455
456        sub_model_names_dict = {"base_estimators": []}
457        for _, estimator in self.base_estimators.items():
458            sub_model_names_dict["base_estimators"].append(estimator.get_model_name())
459
460        sub_model_names_dict["final_estimator"] = [self.final_estimator.get_model_name()]
461
462        return sub_model_names_dict

465class EndaEstimatorWithFallback(EndaEstimator):
466    """
467    This estimator allows to make a prediction even when some important input is missing.
468
469    In order to deal with missing values, it is common practice to replace None/NA with some meaningful value like
470    the mean or the median of the values found in the train set for this feature.
471    However, this is problematic when the missing variable has a significant impact on the prediction.
472
473    Instead, this estimator trains 2 underlying estimators:
474    an 'estimator_with' and an 'estimator_without' the column that can be missing.
475    When predicting, it will use the 'estimator_with' for inputs with the column present
476    and the 'estimator_without' for the others.
477    """
478
479    def __init__(
480            self,
481            resilient_column: str,
482            estimator_with: EndaEstimator,
483            estimator_without: EndaEstimator,
484    ):
485        """
486        Initialize the estimator
487        :param resilient_column: The column that can be missing, which will be taken into account by estimator_with but
488            not by estimator_without
489        :param estimator_with: An EndaEstimator that will train/predict using resilient_column
490        :param estimator_without: An EndaEstimator that will train/predict without using resilient_column
491        """
492
493        if estimator_with is estimator_without:  # check identity
494            raise AttributeError(
495                "estimator_with and estimator_without must be different objects. "
496                "If you want the same base estimator, you can use copying tools "
497                " (like copy.deepcopy()) to duplicate the raw estimator."
498            )
499
500        super().__init__()
501        self.resilient_column = resilient_column
502        self.estimator_with = estimator_with
503        self.estimator_without = estimator_without
504
505    def train(self, df: pd.DataFrame, target_col: str):
506        """
507        Trains the two estimators : estimator_with and estimator_without the resilient_column
508        :param df: The input data for training
509        :param target_col: The column to predict
510        """
511
512        # only train the "estimator_with" where resilient_column is present (not NA)
513        self.estimator_with.train(df.dropna(subset=[self.resilient_column]), target_col)
514        # train the "estimator_without" without resilient_column
515        self.estimator_without.train(
516            df.drop(columns=[self.resilient_column]), target_col
517        )
518
519        # store the training
520        self._training_df = df
521        self._target = target_col
522
523    def predict_both(
524            self, df: pd.DataFrame, target_col: str
525    ) -> (pd.DataFrame, pd.DataFrame):
526        """
527        Makes predictions with both estimators: estimator_with and estimator_without the resilient_column
528        :param df: The input data for prediction
529        :param target_col: The column to predict
530        :return: A tuple with two DataFrames, the first one being the prediction with the resilient_column and the
531            second one without
532        """
533        df_with = df[
534            df[self.resilient_column].notna()
535        ]  # only keeps rows where resilient_column is not NaN
536        prediction_with = self.estimator_with.predict(df_with, target_col)
537        if prediction_with.shape[0] != df.shape[0]:
538            # make estimator_with predict NaN where resilient_column is Nan :
539            prediction_with = prediction_with.reindex(
540                df.index
541            )  # adds rows with NaN values
542
543        if (prediction_with.index != df.index).any():
544            raise ValueError(
545                "prediction_with must have the same index as given df. "
546                "Check that self.estimator_with.predict conserves index."
547            )
548
549        df_without = df.drop(columns=[self.resilient_column])
550        prediction_without = self.estimator_without.predict(df_without, target_col)
551
552        if (prediction_without.index != df.index).any():
553            raise ValueError(
554                "prediction_without must have the same index as given df. "
555                "Check that self.estimator_without.predict conserves index."
556            )
557
558        return prediction_with, prediction_without
559
560    def predict(self, df: pd.DataFrame, target_col: str) -> pd.DataFrame:
561        """
562        Makes a prediction with both estimators, takes prediction by estimator_with by default and fallbacks to
563        estimator_without when it is not available
564        :param df: The input data for prediction
565        :param target_col: The column to predict
566        :return: The final prediction combining both estimators
567        """
