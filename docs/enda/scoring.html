 25def wape(y_true, y_pred, *, sample_weight=None, multioutput="uniform_average"):
 26    """
 27    Weighted average percentage error (WAPE or wMAPE)
 28
 29    Based on code from sklearn.metrics.mean_absolute_percentage_error.
 30    Note here that the output is not a percentage in the range [0, 100].
 31
 32    Parameters
 33    ----------
 34    y_true : array-like of shape (n_samples,) or (n_samples, n_outputs)
 35        Ground truth (correct) target values.
 36
 37    y_pred : array-like of shape (n_samples,) or (n_samples, n_outputs)
 38        Estimated target values.
 39
 40    sample_weight : array-like of shape (n_samples,), default=None
 41        Sample weights.
 42
 43    multioutput : {'raw_values', 'uniform_average'} or array-like
 44        Defines aggregating of multiple output values.
 45        Array-like value defines weights used to average errors.
 46        If input is list then the shape must be (n_outputs,).
 47
 48        'raw_values' :
 49            Returns a full set of errors in case of multioutput input.
 50
 51        'uniform_average' :
 52            Errors of all outputs are averaged with uniform weight.
 53
 54    Returns
 55    -------
 56    loss : float or ndarray of floats
 57        If multioutput is 'raw_values', then mean absolute percentage error
 58        is returned for each output separately.
 59        If multioutput is 'uniform_average' or a ndarray of weights, then the
 60        weighted average of all output errors is returned.
 61    """
 62
 63    y_type, y_true, y_pred, multioutput = _check_reg_targets(
 64        y_true, y_pred, multioutput
 65    )
 66    check_consistent_length(y_true, y_pred, sample_weight)
 67    epsilon = np.finfo(np.float64).eps
 68    mape = np.abs(y_pred - y_true) / np.maximum(np.abs(y_true), epsilon)
 69    custom_weight = y_true / np.maximum(np.sum(y_true), epsilon)  # wMAPE (i.e. WAPE)
 70    if sample_weight is not None:
 71        custom_weight = custom_weight * sample_weight  # wwMAPE
 72    output_errors = np.average(mape, weights=custom_weight, axis=0)
 73    if isinstance(multioutput, str):
 74        if multioutput == "raw_values":
 75            return output_errors
 76        elif multioutput == "uniform_average":
 77            # pass None as weights to np.average: uniform mean
 78            multioutput = None
 79
 80    return np.average(output_errors, weights=multioutput)

102    ):
103        """
104        Initialize the scoring object
105        :param predictions_df: A DataFrame containing the predictions to be scored in columns, as well as a column with the target. The index must be a datetime-index.
106        :param target: The name of the column which contains the target values against which predictions are scored.
107        :param normalizing_col: Optional, a normalizing column for computing normalized absolute error
108        """
109
110        self.predictions_df = predictions_df
111        self.target = target
112        self.normalizing_col = normalizing_col
113        if self.target not in self.predictions_df.columns:
114            raise ValueError(
115                f"target={self.target} must be in predictions_df columns : {self.predictions_df}"

117        if len(self.predictions_df.columns) < 2:
118            raise ValueError(
119                "predictions_df must have at least 2 columns (1 target and 1 prediction)"
120            )
121
122        algo_names = list(
123            c
124            for c in self.predictions_df.columns
125            if c not in [self.target, self.normalizing_col]
126        )

131        error_df = error_df[algo_names]
132        self.error_df = error_df
133
134        self.pct_error_df = (
135            self.error_df.div(self.predictions_df[self.target], axis=0) * 100
136        )

225    @staticmethod
226    def compute_loss(predicted_df: pd.DataFrame,
227                     actual_df: Union[pd.DataFrame, pd.Series],
228                     scores: Union[list[str], dict[str, Callable]] = None) -> pd.Series:
229        """
230        Compute the loss (i.e. the score) between a model prediction and the actual data
231        :param predicted_df: the result of the prediction
232        :param actual_df: the actual target data
233        :param scores: the statistics to consider. Either 'max_error', 'mae', 'rmse', 'r2', 'mape', 'mse', 'wape'.
234            Defaults to 'rmse'.
235            If a list is given, the code detects the statistics to use itself.
236            The score to use can be customized by the user with a dict (name, method).
237        :return: a series that contains for each statistics the score of the model on the training set
238        """
239
240        scores_dict = {}
241        if scores is None:
242            # default is rmse
243            scores_dict = {"rmse": METRICS_FUNCTION_DICT["rmse"]}
244
245        elif isinstance(scores, dict):
246            # home-made scores
247            scores_dict = scores
248
249        elif isinstance(scores, list):
250            # use known functions of enda
251            for score in scores:
252                if score not in METRICS_FUNCTION_DICT:
253                    raise ValueError(f"Score must be one of {METRICS_FUNCTION_DICT.keys()} but got {score}")
254                scores_dict[score] = METRICS_FUNCTION_DICT[score]
255
256        else:
257            raise ValueError("scores must be None, a list of known statistics or a custom home-made "
258                             "dictionary of scores.")
259
260        # we have to compute on the training set the score for each of the method chosen
261        scoring_result_list = []
262        for _, method in scores_dict.items():
263            result_score = method(actual_df, predicted_df)
264            scoring_result_list.append(result_score)
265
266        return pd.Series(data=scoring_result_list, index=scores_dict.keys())

@staticmethod
def compute_loss( predicted_df: pandas.core.frame.DataFrame, actual_df: Union[pandas.core.frame.DataFrame, pandas.core.series.Series], scores: Union[list[str], dict[str, Callable]] = None) -> pandas.core.series.Series: 
226    @staticmethod
227    def compute_loss(predicted_df: pd.DataFrame,
228                     actual_df: Union[pd.DataFrame, pd.Series],
229                     scores: Union[list[str], dict[str, Callable]] = None) -> pd.Series:
230        """
231        Compute the loss (i.e. the score) between a model prediction and the actual data
232        :param predicted_df: the result of the prediction
233        :param actual_df: the actual target data
234        :param scores: the statistics to consider. Either 'max_error', 'mae', 'rmse', 'r2', 'mape', 'mse', 'wape'.
235            Defaults to 'rmse'.
236            If a list is given, the code detects the statistics to use itself.
237            The score to use can be customized by the user with a dict (name, method).
238        :return: a series that contains for each statistics the score of the model on the training set
239        """
240
241        scores_dict = {}
242        if scores is None:
243            # default is rmse
244            scores_dict = {"rmse": METRICS_FUNCTION_DICT["rmse"]}
245
246        elif isinstance(scores, dict):
247            # home-made scores
248            scores_dict = scores
249
250        elif isinstance(scores, list):
251            # use known functions of enda
252            for score in scores:
253                if score not in METRICS_FUNCTION_DICT:
254                    raise ValueError(f"Score must be one of {METRICS_FUNCTION_DICT.keys()} but got {score}")
255                scores_dict[score] = METRICS_FUNCTION_DICT[score]
256
257        else:
258            raise ValueError("scores must be None, a list of known statistics or a custom home-made "
259                             "dictionary of scores.")
260
261        # we have to compute on the training set the score for each of the method chosen
262        scoring_result_list = []
263        for _, method in scores_dict.items():
264            result_score = method(actual_df, predicted_df)
265            scoring_result_list.append(result_score)
266
267        return pd.Series(data=scoring_result_list, index=scores_dict.keys())
